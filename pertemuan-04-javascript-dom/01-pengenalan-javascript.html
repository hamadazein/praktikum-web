Pengenalan JavaScript

1. Apa itu JavaScript?

JavaScript adalah bahasa pemrograman yang digunakan untuk membuat halaman web menjadi interaktif dan dinamis.

Karakteristik JavaScript:

  • Client-side scripting: Berjalan di browser pengguna
  • Dinamis: Dapat mengubah konten HTML secara real-time
  • Event-driven: Merespons aksi pengguna (klik, hover, dll)
  • Interpreted: Tidak perlu dikompilasi

2. Cara Menambahkan JavaScript ke HTML

a. Internal JavaScript (dalam tag <script>)

<script> console.log("Hello from internal JavaScript!"); </script>

b. External JavaScript (file terpisah)

<script src="script.js"></script>

c. Inline JavaScript (dalam atribut HTML)

<button onclick="alert('Hello!')">Click Me</button>

3. JavaScript Dasar

a. Variabel

// ES6+ (Modern JavaScript) let nama = "Ahmad"; // String const umur = 25; // Number (konstanta) var aktif = true; // Boolean // Menampilkan di console console.log("Nama:", nama); console.log("Umur:", umur); console.log("Status:", aktif);

b. Tipe Data

// String let pesan = "Selamat belajar JavaScript!"; let nama = 'Nazriel Rahman Al\'afath'; // Number let angka = 42; let desimal = 3.14; // Boolean let benar = true; let salah = false; // Array let buah = ["apel", "jeruk", "mangga"]; // Object let siswa = { nama: "Nazriel Rahman Al'afath", umur: 20, jurusan: "Informatika" };

c. Operator

// Operator Aritmatika let a = 10; let b = 3; console.log("Penjumlahan:", a + b); // 13 console.log("Pengurangan:", a - b); // 7 console.log("Perkalian:", a * b); // 30 console.log("Pembagian:", a / b); // 3.333... console.log("Modulus:", a % b); // 1 // Operator Perbandingan console.log("a == b:", a == b); // false console.log("a != b:", a != b); // true console.log("a > b:", a > b); // true console.log("a <= b:", a <= b); // false

4. Fungsi (Functions)

a. Function Declaration

function sapa(nama) { return "Halo, " + nama + "!"; } // Memanggil fungsi let hasil = sapa("Nazriel Rahman Al'afath"); console.log(hasil);

b. Function Expression

const perkalian = function(a, b) { return a * b; }; let hasil = perkalian(5, 3); console.log("Hasil:", hasil);

c. Arrow Function (ES6+)

const tambah = (a, b) => a + b; const kuadrat = x => x * x; console.log("Tambah:", tambah(5, 3)); console.log("Kuadrat:", kuadrat(4));

5. Kondisi (If-Else)

let nilai = 85; if (nilai >= 90) { console.log("Grade: A"); } else if (nilai >= 80) { console.log("Grade: B"); } else if (nilai >= 70) { console.log("Grade: C"); } else { console.log("Grade: D"); } // Ternary Operator let status = nilai >= 75 ? "Lulus" : "Tidak Lulus"; console.log("Status:", status);

6. Perulangan (Loops)

a. For Loop

// For loop tradisional for (let i = 1; i <= 5; i++) { console.log("Iterasi ke-" + i); } // For...of (untuk array) let buah = ["apel", "jeruk", "mangga"]; for (let item of buah) { console.log("Buah:", item); }

b. While Loop

let counter = 1; while (counter <= 3) { console.log("Counter:", counter); counter++; }

7. Array Methods

let numbers = [1, 2, 3, 4, 5]; // forEach - iterasi setiap elemen numbers.forEach(function(num) { console.log("Number:", num); }); // map - transformasi array let doubled = numbers.map(num => num * 2); console.log("Doubled:", doubled); // filter - filter elemen let evenNumbers = numbers.filter(num => num % 2 === 0); console.log("Even numbers:", evenNumbers); // reduce - akumulasi nilai let sum = numbers.reduce((total, num) => total + num, 0); console.log("Sum:", sum);

8. Object dan JSON

// Membuat object let mahasiswa = { nama: "Nazriel Rahman Al'afath", nim: "12345678", jurusan: "Teknik Informatika", ipk: 3.75, aktif: true, // Method dalam object perkenalan: function() { return `Halo, saya ${this.nama} dari ${this.jurusan}`; } }; // Mengakses properti console.log("Nama:", mahasiswa.nama); console.log("NIM:", mahasiswa["nim"]); console.log(mahasiswa.perkenalan()); // JSON (JavaScript Object Notation) let jsonString = JSON.stringify(mahasiswa); console.log("JSON:", jsonString); let parsed = JSON.parse(jsonString); console.log("Parsed:", parsed);

9. Console Methods

Berbagai Cara Debugging:

Buka Developer Tools (F12) → Console untuk melihat output

// Berbagai console methods console.log("Pesan biasa"); console.warn("Peringatan"); console.error("Error message"); console.info("Informasi"); // Console dengan styling console.log("%cTeks berwarna!", "color: red; font-size: 20px;"); // Console table let data = [ {nama: "Nazriel Rahman Al'afath", umur: 25}, {nama: "Budi", umur: 23}, {nama: "Citra", umur: 24} ]; console.table(data); // Console group console.group("Data Mahasiswa"); console.log("Nama: Nazriel Rahman Al'afath"); console.log("Jurusan: Informatika"); console.groupEnd();

10. Praktik Terbaik

  • Gunakan let/const instead of var
  • Strict mode: Tambahkan "use strict" di awal script
  • Meaningful names: Gunakan nama variabel yang jelas
  • Consistent formatting: Indentasi dan spacing yang konsisten
  • Comments: Beri komentar pada kode yang kompleks
"use strict"; // Good practices example const CONSTANT_VALUE = 100; let userName = "Nazriel Rahman Al'afath"; let userAge = 25; function calculateUserScore(baseScore, bonus) { // Menghitung skor total pengguna return baseScore + bonus; } // Avoid global variables (function() { // Your code here })();
